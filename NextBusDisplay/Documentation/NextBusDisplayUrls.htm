    DECLARE @train       VARCHAR(30),
        @depart_time INT,
        @dir         VARCHAR(30),
        @stopid      VARCHAR(30),
        @trainid     VARCHAR(50),
        @blockid     VARCHAR(50)
-- Calculate Available Days & Time       
DECLARE @p4 VARCHAR(30) = CASE DATEPART(dw, GETDATE()) WHEN 1 THEN 'Sunday' WHEN 2 THEN 'Monday' WHEN 3 THEN 'Tuesday' WHEN 4 THEN 'Wednesday'
                          WHEN 5 THEN 'Thursday' WHEN 6 THEN 'Friday' WHEN 7 THEN 'Saturday' ELSE '' END
DECLARE @p5 VARCHAR(30) = CASE WHEN @p4 = 'Saturday' OR @p4 = 'Sunday' THEN 'Weekends'
                          ELSE 'Weekdays' END
DECLARE @p6 VARCHAR(30) = CASE WHEN @p5 = 'WeekDays' AND @p4 <> 'Friday' THEN 'Mo-Th' WHEN @p4 = 'Friday' THEN 'Fri-Sun'
                          ELSE 'Fri-Sun' END

SELECT @depart_time = CAST(DATEDIFF(mi, DATEDIFF(d, 0, GETDATE()), GETDATE()) AS INT)

-- Select The Train, Direction & Stop
SELECT @train = 'Sprinter', @dir = 'Oceanside', @stopid = '27006'

-- SELECT @Train = '318', @Dir = 'Oceanside', @StopID = '24346'
-- SELECT @Dir = 'Escondido'
-- 
SELECT @train = '318', @dir = 'Oceanside', @stopid = '22941'

-- 
SELECT @dir = 'Vista'

-- Show The Selections      
SELECT @train train, @dir [direction], @stopid [stop], ( @p4 + ' ' + @p5 + ' ' + @p6 ) [days], @depart_time [depart time]

SELECT -- TOP 2 
        StaticTrainId, Train, Number, Direction, DepartTime,
        CONVERT(VARCHAR, (CAST(DepartTime AS INT)/60)) + ':' + RIGHT('0' + CONVERT(VARCHAR, (CAST(DepartTime AS INT) % 60)), 2) [depart time] ,
        CAST(DepartTime AS INT) - CAST(DATEDIFF(mi, DATEDIFF(d, 0, GETDATE()), GETDATE( ) ) AS INT) [time diff],
        t0.[Platform], [Days], t0.StopID, TrainID, BlockID, t1.[Description]
FROM [dbo].[statictrains] AS [t0]
       INNER JOIN [dbo].[stopdisplay] AS [t1] ON t0.StopID = t1.StopID
WHERE ( [t0].[Train] = @train )
       --  AND ([t0].[Direction] = @Dir)
       AND ( [t0].[StopID] = @stopid )
       AND ( [t0].[Days] IN ( @p4, @p5, @p6 ) )
       AND CAST(DepartTime AS INT) >= @depart_time -- - 20 
--  AND CAST(DepartTime AS INT) <= @Depart_Time + 90
ORDER  BY CONVERT(INT, [t0].[DepartTime])
